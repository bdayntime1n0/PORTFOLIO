Description: ####1# 1# Stock Transfer##
FUNCTION ZFM101_ST.
*"----------------------------------------------------------------------
*"*"Local Interface:
*" IMPORTING
*" REFERENCE(IV_MATNR) TYPE MARA-MATNR
*" REFERENCE(IV_WERKS_FR) TYPE MARC-WERKS
*" REFERENCE(IV_WERKS_TO) TYPE MARC-WERKS
*" REFERENCE(IV_SL_FR) TYPE T001L-LGORT OPTIONAL
*" REFERENCE(IV_SL_TO) TYPE T001L-LGORT OPTIONAL
*" REFERENCE(IV_MENGE) TYPE MSEG-MENGE
*" EXPORTING
*" REFERENCE(EV_RESULT_NO) TYPE BAPI2017_GM_HEAD_RET-MAT_DOC
*" TABLES
*" LT_RETURN STRUCTURE BAPIRET2
*"----------------------------------------------------------------------


* Global data declarations

  DATA : MSGLIN  TYPE C LENGTH 30,
         LV_SLFR LIKE T001L-LGORT,
         LV_SLTO LIKE T001L-LGORT.

  DATA : ST_GOODSMVT_HEADER       LIKE  BAPI2017_GM_HEAD_01,
         ST_GOODSMVT_CODE         LIKE  BAPI2017_GM_CODE,
         ST_GOODSMVT_HEADRET      LIKE  BAPI2017_GM_HEAD_RET,

         G_GOODSMVT_DOC           LIKE  BAPI2017_GM_HEAD_RET-MAT_DOC,
         G_MATERIALDOCUMENT       TYPE  BAPI2017_GM_HEAD_RET-MAT_DOC,
         G_MATDOCUMENTYEAR        TYPE  BAPI2017_GM_HEAD_RET-DOC_YEAR,
         IT_GOODSMVT_ITEM         LIKE  BAPI2017_GM_ITEM_CREATE  OCCURS  0 WITH  HEADER  LINE,
         IT_GOODSMVT_SERIALNUMBER LIKE  BAPI2017_GM_SERIALNUMBER OCCURS  0 WITH  HEADER  LINE.


  DATA L_TABIX LIKE SY-TABIX.


  CLEAR LV_SLFR.
  CLEAR LV_SLTO.

  " #### #### ##
  IF IV_SL_FR IS INITIAL.
    CASE IV_WERKS_FR.
      WHEN '1010'.
        LV_SLFR = '101C'.
      WHEN OTHERS.
        SELECT SINGLE LGORT
         INTO LV_SLFR
         FROM T001L
        WHERE WERKS = IV_WERKS_FR.
    ENDCASE.
  ELSE.
    LV_SLFR = IV_SL_FR.
  ENDIF.

  IF IV_SL_TO IS INITIAL.
    CASE IV_WERKS_TO.
      WHEN '1010'.
        LV_SLTO = '101C'.
      WHEN OTHERS.
        SELECT SINGLE LGORT
         INTO LV_SLTO
         FROM T001L
        WHERE WERKS = IV_WERKS_TO.
    ENDCASE.
  ELSE.
    LV_SLTO = IV_SL_TO.
  ENDIF.



  CLEAR    ST_GOODSMVT_HEADER.
  REFRESH: IT_GOODSMVT_ITEM, IT_GOODSMVT_SERIALNUMBER, LT_RETURN.
  CLEAR:   IT_GOODSMVT_ITEM, IT_GOODSMVT_SERIALNUMBER, LT_RETURN.

  L_TABIX = SY-TABIX.

  CLEAR ST_GOODSMVT_HEADER.
  ST_GOODSMVT_HEADER-PSTNG_DATE   = SY-DATUM. " ####
  ST_GOODSMVT_HEADER-DOC_DATE     = SY-DATUM. " ####
  ST_GOODSMVT_CODE-GM_CODE        = '04'.     " ####

  REFRESH: IT_GOODSMVT_ITEM, IT_GOODSMVT_SERIALNUMBER, LT_RETURN.
  CLEAR:   IT_GOODSMVT_ITEM, IT_GOODSMVT_SERIALNUMBER, LT_RETURN.

  IT_GOODSMVT_ITEM-MOVE_TYPE  = '301'.

  IT_GOODSMVT_ITEM-MATERIAL   = IV_MATNR.     " ####
  IT_GOODSMVT_ITEM-PLANT      = IV_WERKS_FR.  " ### (### #)
  IT_GOODSMVT_ITEM-STGE_LOC   = LV_SLFR.      " S/L

  IT_GOODSMVT_ITEM-MOVE_MAT   = IV_MATNR.    " ####
  IT_GOODSMVT_ITEM-MOVE_PLANT = IV_WERKS_TO.  " ### ### (## #)
  IT_GOODSMVT_ITEM-MOVE_STLOC = LV_SLTO.      "  S/L

  IT_GOODSMVT_ITEM-ENTRY_QNT  = IV_MENGE.     " ##
  IT_GOODSMVT_ITEM-ENTRY_UOM  = 'EA'.
  APPEND IT_GOODSMVT_ITEM.


  CLEAR : G_MATERIALDOCUMENT, G_MATDOCUMENTYEAR.
  CALL FUNCTION 'BAPI_GOODSMVT_CREATE'
    EXPORTING
      GOODSMVT_HEADER  = ST_GOODSMVT_HEADER
      GOODSMVT_CODE    = ST_GOODSMVT_CODE
    IMPORTING
      GOODSMVT_HEADRET = ST_GOODSMVT_HEADRET
      MATERIALDOCUMENT = G_MATERIALDOCUMENT
      MATDOCUMENTYEAR  = G_MATDOCUMENTYEAR
    TABLES
      GOODSMVT_ITEM    = IT_GOODSMVT_ITEM
      RETURN           = LT_RETURN.


  CLEAR G_GOODSMVT_DOC.
  IF NOT ST_GOODSMVT_HEADRET IS INITIAL.
    G_GOODSMVT_DOC = ST_GOODSMVT_HEADRET-MAT_DOC.
    CALL FUNCTION 'BAPI_TRANSACTION_COMMIT'.
    EV_RESULT_NO =  G_GOODSMVT_DOC.
  ELSE.
    READ TABLE LT_RETURN  INDEX 1.
      MSGLIN = LT_RETURN-MESSAGE.
  ENDIF.

ENDFUNCTION.
