FUNCTION ZBD101_PO.
*"----------------------------------------------------------------------
*"*"Local Interface:
*" IMPORTING
*" REFERENCE(G_BUKRS) TYPE EKKO-BUKRS
*" REFERENCE(G_BSART) TYPE EKKO-BSART
*" REFERENCE(G_LIFNR) TYPE EKKO-LIFNR
*" REFERENCE(G_EKORG) TYPE EKKO-EKORG
*" REFERENCE(G_EKGRP) TYPE EKKO-EKGRP
*" REFERENCE(LS_MATNR) TYPE MARA-MATNR
*" REFERENCE(LS_WERKS) TYPE MARC-WERKS
*" REFERENCE(LS_MENGE) TYPE MSEG-MENGE

*"----------------------------------------------------------------------


* Global data declarations

DATA : LS_POHEADER LIKE BAPIMEPOHEADER,
       LS_POHEADERX LIKE  BAPIMEPOHEADERX,
* LT_RETURN LIKE TABLE OF BAPIRET2 WITH HEADER LINE,
       LT_POITEM    LIKE  TABLE OF BAPIMEPOITEM  WITH HEADER LINE,
       LT_POITEMX   LIKE  TABLE OF BAPIMEPOITEMX WITH HEADER LINE.
* LV_ORDER LIKE BAPIMEPOHEADER-PO_NUMBER.

DATA : LV_PRICE LIKE MBEW-STPRS.

SELECT SINGLE STPRS
  INTO LV_PRICE
  FROM MBEW
 WHERE MATNR = LS_MATNR.


CLEAR : LS_POHEADER.
LS_POHEADER-COMP_CODE = G_BUKRS. " ####
LS_POHEADER-DOC_TYPE = G_BSART. " ####
LS_POHEADER-VENDOR = G_LIFNR. " Vendor
LS_POHEADER-PURCH_ORG = G_EKORG. " ####
LS_POHEADER-PUR_GROUP = G_EKGRP. " ####

CLEAR : LS_POHEADERX.
LS_POHEADERX-COMP_CODE = 'X'.
LS_POHEADERX-DOC_TYPE = 'X'.
LS_POHEADERX-VENDOR = 'X'.
LS_POHEADERX-PURCH_ORG = 'X'.
LS_POHEADERX-PUR_GROUP = 'X'.

CLEAR : LT_POITEM, LT_POITEM[].
LT_POITEM-PO_ITEM = '00010'.
LT_POITEM-MATERIAL = LS_MATNR. " ##
LT_POITEM-PLANT = LS_WERKS. " ###
LT_POITEM-QUANTITY = LS_MENGE. " ##
*LT_POITEM-NET_PRICE = LS_PRICE. " ##
LT_POITEM-NET_PRICE = LV_PRICE. " ##
APPEND LT_POITEM.

CLEAR : LT_POITEMX, LT_POITEMX[].
LT_POITEMX-PO_ITEM = '00010'.
LT_POITEMX-MATERIAL = 'X'.
LT_POITEMX-PLANT = 'X'.
LT_POITEMX-QUANTITY = 'X'.
LT_POITEMX-NET_PRICE = 'X'.
APPEND LT_POITEMX.

CLEAR LV_ORDER.
CLEAR : LT_RETURN, LT_RETURN[].
CALL FUNCTION 'BAPI_PO_CREATE1'
  EXPORTING
    POHEADER         = LS_POHEADER
    POHEADERX        = LS_POHEADERX
  IMPORTING
    EXPPURCHASEORDER = LV_ORDER
  TABLES
    RETURN           = LT_RETURN
    POITEM           = LT_POITEM
    POITEMX          = LT_POITEMX.

IF LV_ORDER IS NOT INITIAL.
  CALL FUNCTION 'BAPI_TRANSACTION_COMMIT'.


  WRITE :  LV_ORDER.
ELSE.
  CALL FUNCTION 'BAPI_TRANSACTION_ROLLBACK'.

  WRITE :/ LT_RETURN[ TYPE = 'E' ]-MESSAGE.

ENDIF.



ENDFUNCTION.
