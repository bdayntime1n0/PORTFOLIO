MODULE EXIT_0100 INPUT.
  SAVEOK = OK_CODE.
  CLEAR OK_CODE.

  CASE SAVEOK.
    WHEN 'EXIT'. LEAVE PROGRAM.
    WHEN 'CANC' OR 'BACK'. LEAVE TO SCREEN 0.
  ENDCASE.
ENDMODULE.

MODULE USER_COMMAND_0100 INPUT.

 SAVEOK = OK_CODE.
 CLEAR OK_CODE.

 CASE SAVEOK.
   WHEN 'SAVE'.
    PERFORM CHK_SAVE.
   WHEN 'CHANGE'.
    PERFORM CHANGE_MODE.
   WHEN OTHERS.
 ENDCASE.

ENDMODULE.
