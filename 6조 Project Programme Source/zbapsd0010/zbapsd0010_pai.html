MODULE USER_COMMAND_0100 INPUT.

  SAVEOK = OKCODE. CLEAR OKCODE.

  CASE SAVEOK.
    WHEN 'BACK'.  LEAVE TO SCREEN 0.
*    WHEN 'SAVEDATA'. perform save_data.
    when 'ALLCREATE'. PERFORM create_all_so.
    WHEN 'P_CHAN'. PERFORM CHAN_MODE.
    WHEN 'P_DISP'. perform save_data.
*                   PERFORM DISP_MODE.
*                   PERFORM TOOLBAR.
    WHEN 'SELCREATE'.  PERFORM SELECT_SAVE_DATA.

    WHEN OTHERS.
  ENDCASE.

ENDMODULE.

MODULE EXIT_COMMAND_0100 INPUT.

  CLEAR OKCODE.

  CASE SAVEOK.
    WHEN 'EXIT'.
      LEAVE PROGRAM.

    WHEN 'CANC'.
      LEAVE TO SCREEN 0.

    WHEN OTHERS.
  ENDCASE.
ENDMODULE.
