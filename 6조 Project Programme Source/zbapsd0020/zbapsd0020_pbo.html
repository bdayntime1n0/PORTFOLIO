          CHECK S_DELDT[] IS INITIAL.
          S_DELDT-SIGN = 'I'.
          S_DELDT-OPTION = 'BT'.
          CONCATENATE SY-DATUM(6)'01' INTO S_DELDT-LOW.

          PERFORM LAST_DATE_OF_MONTH USING S_DELDT-LOW
                                  CHANGING S_DELDT-HIGH.

          APPEND S_DELDT. CLEAR S_DELDT.

        ELSE.
          SCREEN-INPUT = 0.
        ENDIF.

*..   ####
      WHEN 'Z2'.

                                  CHANGING S_AUDAT-HIGH.

        ELSE.
          SCREEN-INPUT = 0.
        ENDIF.

          SCREEN-INVISIBLE = 1.
        ELSEIF P_B2C = 'X'.
          SCREEN-INVISIBLE = 0.
        ENDIF.

      WHEN OTHERS.
    ENDCASE.

    MODIFY SCREEN.
  ENDLOOP.
