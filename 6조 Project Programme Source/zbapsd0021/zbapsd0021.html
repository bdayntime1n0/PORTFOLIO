REPORT ZBAPSD0021.
TABLES : VBAK,VBAP,MAKT.

*DATA:BEGIN OF GS_DISPLAY,
*       KUNNR LIKE VBAK-KUNNR,    "SOLD TO PARTY
*       BSTNK LIKE VBAK-BSTNK,    "##ID
**       BNAME LIKE VBAK-BNAME,    "#####
*       MATNR LIKE VBAP-MATNR,    "MATERIAL NUMBER
*       MAKTX LIKE MAKT-MAKTX,    "## ##
*       AUDAT LIKE VBAK-AUDAT,    "###
*       VDATU LIKE VBAK-VDATU,    "###
*
*     END OF GS_DISPLAY.
DATA:BEGIN OF GS_SUB_INFO,
       ORDERER_ID LIKE VBAK-KUNNR,    "SOLD TO PARTY
       SUB_ID     LIKE VBAK-BSTNK,    "##ID
*       BNAME LIKE VBAK-BNAME,    "#####
       MAT_ID     LIKE VBAP-MATNR,    "MATERIAL NUMBER
       MAT_NAME   LIKE MAKT-MAKTX,    "## ##
       ORDER_DATE LIKE VBAK-AUDAT,    "###
       DEL_DATE   LIKE VBAK-VDATU,    "###
*       ETENR      LIKE VBEP-ETENR,    "#####
     END OF GS_SUB_INFO.

DATA GT_SUB_INFO LIKE TABLE OF GS_SUB_INFO.
PARAMETERS: P_KUNNR TYPE VBAK-KUNNR.
PARAMETERS: P_BSTNK TYPE VBAK-BSTNK.
*PARAMETERS: P_SUBST TYPE VBAK-VDATU.
SELECT DISTINCT A~KUNNR,
       A~BSTNK,
*       A~BNAME
       B~MATNR,
       C~MAKTX,
       A~AUDAT,
       A~VDATU
*       D~ETENR
  INTO TABLE @GT_SUB_INFO
  FROM VBAK AS A JOIN VBAP AS B
                 ON B~VBELN = A~VBELN
                 INNER JOIN MAKT AS C
                   ON B~VBELN = A~VBELN
                 AND B~MATNR = C~MATNR
*                INNER JOIN VBEP AS D
*                          ON  B~VBELN EQ D~VBELN
*                         and  D~etenr eq '0001'
  WHERE A~BSTNK = @P_BSTNK
  AND   A~KUNNR = @P_KUNNR
  ORDER BY VDATU.

CL_DEMO_OUTPUT=>DISPLAY( GT_SUB_INFO ).
