REPORT ZSG_BW_DL.

TABLES ZVEVAL.

DATA LS_VEVAL LIKE ZVEVAL.
DATA LT_VEVAL LIKE TABLE OF LS_VEVAL.

REFRESH LT_VEVAL.

  SELECT DISTINCT A~VBELN,  " ####
                  A~VDATU,  " #####
                  D~WADAT,  " ###
                  C~LIFNR,  " ##
                  B~MATNR,  " ##
                  B~ARKTX,  " ###
                  A~NETWR   " ###
    FROM VBAK AS A JOIN VBAP AS B ON A~VBELN = B~VBELN
                   JOIN EINA AS C ON C~MATNR = B~MATNR
                   JOIN VBEP AS D ON A~VBELN = D~VBELN
    WHERE C~LIFNR BETWEEN 'SG_BP01_I' AND 'SG_BP10_I'
    ORDER BY C~LIFNR, A~VBELN
    INTO TABLE @LT_VEVAL.

LOOP AT LT_VEVAL INTO LS_VEVAL.
  IF LS_VEVAL-V_GID > LS_VEVAL-V_RDD.
    LS_VEVAL-V_OPN = 'X'.
    LS_VEVAL-V_DLP = 0.
    MODIFY LT_VEVAL FROM LS_VEVAL.
  ELSE.
    LS_VEVAL-V_DLP = 1.
    MODIFY LT_VEVAL FROM LS_VEVAL.
  ENDIF.
ENDLOOP.

CL_DEMO_OUTPUT=>DISPLAY( LT_VEVAL ).

DELETE FROM ZVEVAL.
MODIFY ZVEVAL FROM TABLE LT_VEVAL. COMMIT WORK.
