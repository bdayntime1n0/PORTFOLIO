REPORT ZSG_BW_BG.
TABLES: ZVEVAL, ZVEVAL_G.

DATA LS_VEVAL LIKE ZVEVAL.
DATA LT_VEVAL LIKE TABLE OF LS_VEVAL.

DATA LS_VEVAL_G LIKE ZVEVAL_G.
DATA LT_VEVAL_G LIKE TABLE OF ZVEVAL_G.

DATA: CHECK_G TYPE P DECIMALS 2,
      TOT TYPE P DECIMALS 2,
      COM TYPE P DECIMALS 2.

* 1## V_RDD BETWEEN '20210101' AND '20210331'
* 2## V_RDD BETWEEN '20210401' AND '20210630'
* 3## V_RDD BETWEEN '20210701' AND '20210930'
* 4## V_RDD BETWEEN '20211001' AND '20211231'

REFRESH LT_VEVAL.
REFRESH LT_VEVAL_G.

SELECT V_VEN AS G_VEN ,COUNT( V_SO ) AS G_TOT, SUM( V_DLP ) AS G_COM"
  FROM ZVEVAL
 WHERE V_RDD BETWEEN '20210701' AND '20210930'
   AND V_GID BETWEEN '20210701' AND '20210930'
 GROUP BY V_VEN
  INTO CORRESPONDING FIELDS OF TABLE @LT_VEVAL_G.

LOOP AT LT_VEVAL_G INTO LS_VEVAL_G.
  IF LS_VEVAL_G-G_QUT IS INITIAL.
    LS_VEVAL_G-G_QUT = '2021## 3##'.  " ### - ##### ## ####
    CLEAR CHECK_G.
    COM = LS_VEVAL_G-G_COM. TOT = LS_VEVAL_G-G_TOT.
    CHECK_G = COM / TOT * 100.
      IF CHECK_G > 90.
        LS_VEVAL_G-G_GRD = 'A'.
      ELSEIF CHECK_G > 75.
        LS_VEVAL_G-G_GRD = 'B'.
      ELSEIF CHECK_G > 60.
        LS_VEVAL_G-G_GRD = 'C'.
      ELSE.
        LS_VEVAL_G-G_GRD = 'D'.
      ENDIF.
    MODIFY LT_VEVAL_G FROM LS_VEVAL_G.
  ENDIF.
ENDLOOP.

CL_DEMO_OUTPUT=>DISPLAY( LT_VEVAL_G ).

*DELETE FROM ZVEVAL_G.
MODIFY ZVEVAL_G FROM TABLE LT_VEVAL_G. COMMIT WORK.
