FORM SELECT_SO.
  REFRESH GT_SODISP.

  SELECT DISTINCT B~WERKS,  " Plant
                  A~VBELN,  " SO Number
                  A~ERDAT,  " SO Created Date
                  A~AUDAT,  " SO Doc Date
                  C~LIFNR,  " Vendor
                  B~MATNR,  " Material
                  B~ARKTX,  " Material Desc.
                  A~VKORG,  " Sales Organization
                  A~KUNNR,  " Sold-To-Party
                  A~BSTNK,  " Customer Ref.
                  B~KWMENG, " Order Quantity
*                  B~LSMENG, " Required Delivery Qty
                  B~KBMENG, " Confirmed Quantity in Sales Unit
                  A~NETWR,  " Net Value
                  B~NETPR,  " Net Price
                  A~WAERK,  " Currency
                  E~BANFN,  " Purchasing Req
                  D~EKORG   " Purchasing organization
    FROM VBAK AS A JOIN VBAP AS B ON A~VBELN = B~VBELN
                   JOIN EINA AS C ON C~MATNR = B~MATNR AND C~LOEKZ <> 'X'
                   JOIN LFM1 AS D ON C~LIFNR = D~LIFNR
                   JOIN VBEP AS E ON A~VBELN = E~VBELN
   WHERE A~KUNNR IN @S_CUST
     AND A~KUNNR LIKE 'SG_BC_%'  "# ###
     AND B~MATNR IN @S_MATC
     AND B~MATNR LIKE 'SG_A_%'   "# ###
     AND A~ERDAT IN @S_ODAT
   ORDER BY A~VBELN
    INTO CORRESPONDING FIELDS OF TABLE @GT_SODISP.
ENDFORM.

FORM PO_BAPI.

  DATA LT_ROW TYPE LVC_T_ROW. " ## ### ## ### ## Internal Table
  DATA LS_ROW TYPE LVC_S_ROW.

  DATA LV_REFRESH TYPE C.     " ALV # Refresh ### #### Flag ##
  DATA LV_LINES TYPE I.

  CALL METHOD GO_ALV_GRID->GET_SELECTED_ROWS
    IMPORTING
      ET_INDEX_ROWS = LT_ROW.

  DESCRIBE TABLE LT_ROW LINES LV_LINES.
  IF LV_LINES = 0.
    MESSAGE 'ROW# ######!' TYPE 'S' DISPLAY LIKE 'W'.
    EXIT.
  ENDIF.

  LOOP AT LT_ROW INTO LS_ROW.

    READ TABLE GT_SODISP INTO GS_SO INDEX LS_ROW-INDEX.
    IF SY-SUBRC <> 0.
      EXIT.
    ENDIF.

    CLEAR : LS_POHEADER.
    LS_POHEADER-COMP_CODE  = '1010'.
    LS_POHEADER-DOC_TYPE   = 'NB'.
    LS_POHEADER-VENDOR     = GS_SO-LIFNR.
    LS_POHEADER-PURCH_ORG  = GS_SO-EKORG.
    LS_POHEADER-PUR_GROUP  = '001'.
    LS_POHEADER-CREAT_DATE = GS_SO-AUDAT.
    LS_POHEADER-DOC_DATE = GS_SO-AUDAT.


    CLEAR : LS_POHEADERX.
    LS_POHEADERX-COMP_CODE  =  ABAP_TRUE.
    LS_POHEADERX-DOC_TYPE   =  ABAP_TRUE.
    LS_POHEADERX-VENDOR     =  ABAP_TRUE.
    LS_POHEADERX-PURCH_ORG  =  ABAP_TRUE.
    LS_POHEADERX-PUR_GROUP  =  ABAP_TRUE.
    LS_POHEADERX-CREAT_DATE =  ABAP_TRUE.
    LS_POHEADERX-DOC_DATE   =  ABAP_TRUE.


    CLEAR : LT_POITEM, LT_POITEM[].
    LT_POITEM-PO_ITEM   = '00010'.
    LT_POITEM-MATERIAL  = GS_SO-MATNR.
    LT_POITEM-PLANT     = GS_SO-WERKS.
    LT_POITEM-QUANTITY  = GS_SO-KWMENG.
    LT_POITEM-PREQ_NO   = GS_SO-BANFN.
    APPEND LT_POITEM.

    CLEAR : LT_POITEMX, LT_POITEMX[].
    LT_POITEMX-PO_ITEM   = '00010'.
    LT_POITEMX-MATERIAL  = ABAP_TRUE.
    LT_POITEMX-PLANT     = ABAP_TRUE.
    LT_POITEMX-QUANTITY  = ABAP_TRUE.
    LT_POITEMX-PREQ_NO   = ABAP_TRUE.
    APPEND LT_POITEMX.

    CLEAR LV_ORDER.
    CLEAR : LT_RETURN, LT_RETURN[].
    CALL FUNCTION 'BAPI_PO_CREATE1'
      EXPORTING
        POHEADER         = LS_POHEADER
        POHEADERX        = LS_POHEADERX
      IMPORTING
        EXPPURCHASEORDER = LV_ORDER
      TABLES
        RETURN           = LT_RETURN
        POITEM           = LT_POITEM
        POITEMX          = LT_POITEMX.

    IF LV_ORDER IS NOT INITIAL.
      CALL FUNCTION 'BAPI_TRANSACTION_COMMIT'.
      GS_SO-STATUS = ICON_GREEN_LIGHT.
      GS_SO-PONUM  = LV_ORDER.
      MODIFY GT_SODISP FROM GS_SO INDEX LS_ROW-INDEX.
    ELSE.
      CALL FUNCTION 'BAPI_TRANSACTION_ROLLBACK'.
      GS_SO-STATUS = ICON_RED_LIGHT.
      MODIFY GT_SODISP FROM GS_SO INDEX LS_ROW-INDEX.
    ENDIF.

  ENDLOOP.

  PERFORM REFRESH_ALV_GRID.
ENDFORM.
