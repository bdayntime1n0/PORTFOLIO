MODULE exit_0100 INPUT.
    SAVEOK = OKCODE. CLEAR OKCODE.

  CASE SAVEOK.
    WHEN 'BACK'.  LEAVE TO SCREEN 0.
    WHEN 'EXIT'.  LEAVE PROGRAM.
    WHEN 'CANC'.  LEAVE TO SCREEN 0.
  ENDCASE.
ENDMODULE.

MODULE user_command_0100 INPUT.

    SAVEOK = OKCODE. CLEAR OKCODE.

  CASE SAVEOK.
    WHEN 'RESO'.
      PERFORM select_so.
      GR_GRID_DATA->refresh_table_display(
*        EXPORTING
*          is_stable      =                  " With Stable Rows/Columns
*          i_soft_refresh =                  " Without Sort, Filter, etc.
*        EXCEPTIONS
*          finished       = 1                " Display was Ended (by Export)
*          others         = 2
      ).
      IF SY-SUBRC <> 0.
*       MESSAGE ID SY-MSGID TYPE SY-MSGTY NUMBER SY-MSGNO
*         WITH SY-MSGV1 SY-MSGV2 SY-MSGV3 SY-MSGV4.
      ENDIF.

    WHEN 'DISO'.
      GR_GRID_DATA->get_selected_rows(
        IMPORTING
*          et_index_rows =                  " Indexes of Selected Rows
          et_row_no     =  DATA(LT_ROWS)                " Numeric IDs of Selected Rows
      ).
      READ TABLE LT_ROWS INTO DATA(LS_ROWS) INDEX 1.
      CHECK SY-SUBRC = 0.
      READ TABLE GT_SOJOIN INTO DATA(LS_SOJOIN) INDEX LS_ROWS-row_id.
      CHECK SY-SUBRC = 0.
      SET PARAMETER ID 'AUN' FIELD LS_SOJOIN-vbeln.
      SET UPDATE TASK LOCAL.
      CALL TRANSACTION 'VA02' AND SKIP FIRST SCREEN.
    WHEN 'CREATE'.
        PERFORM SO_CREATE.
  ENDCASE.

ENDMODULE.
