FORM SELECT_PO.
  REFRESH GT_PODISP.

  SELECT A~EBELN,  " Puchasing Order
         A~EKORG,  " Puchasing Organization
         A~EKGRP,  " Puchasing Group
         A~LIFNR,  " Vender
         B~EBELP,  " Item
         B~MATNR,  " Material
         B~TXZ01,  " Material Desc.
         B~WERKS,  " Plant
*         B~LGORT,  " Storage location
         B~MENGE,  " Quantity
         B~MEINS,  " Unit of Measure
         B~NETPR,  " Net Price
         B~INFNR,  " Info Record
         A~AEDAT,  " PO Created Date
         A~ERNAM,  " PO Created Person
         C~VBELN   " SO Number
*         A~LOEKZ    " Delete Indicator
    FROM EKKO AS A JOIN EKPO AS B
                     ON A~EBELN = B~EBELN
                   JOIN VBEP AS C
                     ON B~BANFN = C~BANFN
                    AND B~BNFPO = C~BNFPO
   WHERE B~WERKS IN @P_PLNT
     AND B~MATNR IN @P_MATC
     AND B~MATNR LIKE 'SG_A_%'
     AND B~BANFN <> ' '
     AND A~LIFNR IN @P_VEND
     AND B~LOEKZ NOT LIKE 'L'
*     AND A~AEDAT IN @P_ODAT
   ORDER BY A~EBELN
    INTO TABLE @GT_PODISP.
ENDFORM.
