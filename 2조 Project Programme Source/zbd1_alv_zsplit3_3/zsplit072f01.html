FORM SELECT_DATA2 USING PV_LIFNR.

* # ## ALV ## DATA SELECT
  SELECT
      A~MATNR
    , B~MAKTX
    , SUM( A~MENGE ) AS MENGE
    , SUM( DISTINCT C~LABST ) AS LABST
    FROM ZTMM_13 AS A
    LEFT OUTER JOIN MAKT AS B
                 ON A~MATNR EQ B~MATNR
         INNER JOIN MARD AS C
                 ON A~MATNR EQ C~MATNR
    INTO CORRESPONDING FIELDS OF TABLE @GT_DISP2
   WHERE A~LIFNR EQ @PV_LIFNR
    AND A~BUDAT IN @S_BUDAT[]
    AND A~MATNR IN @S_MATNR
    GROUP BY A~MATNR, B~MAKTX
    ORDER BY A~MATNR.

ENDFORM.
