FORM SELECTED_DATA_RTN .

  REFRESH GT_DISPLAY.


  SELECT
           VBAK~ERDAT                         AS  SODAT      ,
           VBAK~VBELN                         AS  SONUM      ,
           BUT000~NAME_ORG1                   AS  NMSORT     , "##/##
           KNA1~KUNNR                         AS  CUSTID     , "BP ##
           BUT000~NAME_ORG2                   AS  CUSTNM     , "####
           MAKT~MAKTX                         AS  CARNM      , "###
           TSPAT~VTEXT                       AS  CARSO      ,  "##
           VBAP~KWMENG                        AS  SOQTY      ,
           VBAP~NETWR                         AS  SOSUM


  FROM  VBAK    INNER JOIN VBAP ON VBAK~VBELN EQ VBAP~VBELN
                INNER JOIN BUT000 ON VBAK~KUNNR EQ BUT000~PARTNER
                INNER JOIN KNA1 ON VBAK~KUNNR EQ KNA1~KUNNR
                LEFT OUTER JOIN MAKT ON VBAP~MATNR EQ MAKT~MATNR
                                AND MAKT~SPRAS EQ @SY-LANGU
                LEFT OUTER JOIN TSPAT ON VBAK~SPART EQ TSPAT~SPART
                                      AND TSPAT~SPRAS EQ @SY-LANGU

   " Selection Screen ## ###### #### ####.
  WHERE VBAK~ERDAT IN @S_SODAT
    AND VBAK~VBELN IN @S_SONUM
    AND MAKT~MAKTX IN @S_CARNM
    AND TSPAT~VTEXT IN @S_CARSO
    AND BUT000~NAME_ORG1 IN @S_NMSORT
    AND BUT000~NAME_ORG2 IN @S_CUSTNM

    INTO TABLE @GT_DISPLAY.

  IF SY-SUBRC EQ 0.
    SORT GT_DISPLAY BY SODAT
                       SONUM
                       CARNM.

  ENDIF.
