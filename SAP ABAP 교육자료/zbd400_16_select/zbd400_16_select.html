REPORT ZBD400_16_SELECT.

PARAMETERS : P_CARRID TYPE SFLIGHT-CARRID.

START-OF-SELECTION.

  DATA : LT_SFLIGHT TYPE TABLE OF SFLIGHT,   "INTERNAL TABLE
         LS_SFLIGHT LIKE LINE OF LT_SFLIGHT. "STRUCTIORE

  SELECT CARRID
         CONNID
         FLDATE
         SEATSMAX
         SEATSOCC
    FROM SFLIGHT
    INTO CORRESPONDING FIELDS OF TABLE LT_SFLIGHT
    WHERE CARRID = P_CARRID.


  IF SY-SUBRC = 0.
    LOOP AT LT_SFLIGHT INTO LS_SFLIGHT.
      WRITE : / LS_SFLIGHT-CARRID,
                LS_SFLIGHT-CONNID,
                LS_SFLIGHT-FLDATE,
                LS_SFLIGHT-SEATSMAX,
                LS_SFLIGHT-SEATSOCC.
    ENDLOOP.
  ELSE.
    WRITE : / 'NO DATA FOUND.'.
  ENDIF.
