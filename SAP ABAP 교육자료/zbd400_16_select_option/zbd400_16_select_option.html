REPORT ZBD400_16_SELECT_OPTION.

TABLES SCARR. " DATA SCARR TYPE SCARR 과 동일.

PARAMETERS : P_CARRID TYPE SCARR-CARRID.

" SELECT-OPTIONS는 DICTIONARY 변수 참조가 불가능.
SELECT-OPTIONS : S_CARRID FOR SCARR-CARRID.


START-OF-SELECTION.

  DATA : LT_SPFLI TYPE TABLE OF SPFLI,
         LS_SPFLI LIKE LINE OF LT_SPFLI.

  SELECT *
      FROM SPFLI
      INTO TABLE LT_SPFLI
      WHERE CARRID IN S_CARRID.

  IF SY-SUBRC = 0.
    LOOP AT LT_SPFLI INTO LS_SPFLI.
      WRITE : / LS_SPFLI-CARRID,
                LS_SPFLI-CONNID,
                LS_SPFLI-CITYFROM,
                LS_SPFLI-CITYTO.
    ENDLOOP.

  ELSE.
    WRITE : /'NO DATA FOUND.'.

  ENDIF.
