REPORT ZBD400_16_SELECT_SINGLE_3.

PARAMETERS : P_R1 RADIOBUTTON GROUP RAG1 DEFAULT 'X',
             P_R2 RADIOBUTTON GROUP RAG1,
             P_R3 RADIOBUTTON GROUP RAG1.

** 전역 변수
DATA : GV_MAX TYPE SFLIGHT-SEATSMAX,
       GV_OCC    TYPE SFLIGHT-SEATSOCC,
       GS_STRUCT TYPE SFLIGHT.

START-OF-SELECTION.

  CASE 'X'.
    WHEN P_R1.
      SELECT SINGLE SEATSMAX SEATSOCC
        FROM SFLIGHT INTO (GV_MAX, GV_OCC)
        WHERE CARRID = 'AA'
          AND CONNID = '0017'
          AND FLDATE ='20210920'.

    WHEN P_R2.
        SELECT SINGLE SEATSMAX SEATSOCC
        FROM SFLIGHT
          INTO (GS_STRUCT-SEATSMAX, GS_STRUCT-SEATSOCC)
        WHERE CARRID = 'AA'
          AND CONNID = '0017'
          AND FLDATE ='20210920'.

    WHEN P_R3.
         SELECT SINGLE *
        FROM SFLIGHT
        INTO CORRESPONDING FIELDS OF GS_STRUCT
        WHERE CARRID = 'AA'
        AND CONNID = '0017'
        AND FLDATE = '20210920'.

      SELECT SINGLE SEATSMAX SEATSOCC
        FROM SFLIGHT
        INTO CORRESPONDING FIELDS OF GS_STRUCT
        WHERE CARRID = 'AA'
        AND CONNID = '0017'
        AND FLDATE = '20210920'.
*이 경우 SEATSMAX, SEATSOCC는 OVERWRITE됨.

  ENDCASE.


***  DISPLAY

  WRITE /: GV_MAX,
           GV_OCC,
           GS_STRUCT-SEATSMAX,
           GS_STRUCT-SEATSOCC.
