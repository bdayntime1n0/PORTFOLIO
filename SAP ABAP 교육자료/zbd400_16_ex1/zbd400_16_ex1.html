REPORT ZBD400_16_EX1.

* 전재목

PARAMETERS : PA_LOW TYPE NUM2,
             PA_HIGH TYPE NUM2.

DATA : LV_RESULT TYPE NUM2,
       LV_WRITE  TYPE NUM2,
       LV_NUM    TYPE NUM2,
       LV_TIMES  TYPE NUM2.

IF PA_LOW < 2.
  WRITE 'PA_LOW 는 2 보다 작을 수 없습니다.'.

ELSEIF PA_HIGH > 9.
  WRITE 'PA_HIGH 는 9 보다 클 수 없습니다.'.

ELSEIF PA_HIGH = 0.

  DO 9 TIMES.
    LV_NUM = SY-INDEX.
    LV_RESULT = PA_LOW * LV_NUM.
    WRITE : / PA_LOW, ' * ', LV_NUM, ' = ', LV_RESULT.
  ENDDO.

ELSEIF PA_LOW <= PA_HIGH.

  LV_TIMES = PA_HIGH - PA_LOW + 1.

  DATA : LV_DAN TYPE NUM2.

  DO 9 TIMES.
    LV_DAN = PA_LOW.
    LV_NUM = LV_NUM + 1.

    DO LV_TIMES TIMES.
      LV_RESULT = LV_DAN * LV_NUM.
      WRITE : AT LV_WRITE LV_DAN, '*', LV_NUM, '=', LV_RESULT.
      LV_WRITE = LV_WRITE + 30.
      LV_DAN = LV_DAN + 1.
    ENDDO.

    CLEAR LV_WRITE.
    SKIP 1.

  ENDDO.

ELSEIF PA_HIGH < PA_LOW.
  WRITE 'LOW 는 HIGH 보다 작거나 같아야 합니다.'.

ENDIF.
