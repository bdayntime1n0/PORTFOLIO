FORM SELECT_DATA.

  REFRESH GT_SPFLI.

  SELECT *
    FROM ZBD1_PFLI16
    INTO TABLE @GT_SPFLI
    WHERE CARRID IN @S_CARRID
      AND CONNID IN @S_CONNID
    ORDER BY PRIMARY KEY.

ENDFORM.

*& Form SAVE_DATA
*&---------------------------------------------------------------------*
FORM SAVE_DATA .

" GV_MODE = A : 생성. B : 수정 C : 삭제 D : 조회
  CASE GV_MODE.
    WHEN 'A'.   " 생성
      IF ZBD1_PFLI16-CARRID IS INITIAL
         OR ZBD1_PFLI16-CONNID IS INITIAL.
        MESSAGE S000(OO) WITH '모든 필수입력 필드에 값을 입력하세요.' DISPLAY LIKE 'E'.
        EXIT.
      ENDIF.

      SELECT SINGLE *
        FROM ZBD1_PFLI16
        INTO CORRESPONDING FIELDS OF ZBD1_PFLI16
        WHERE CARRID = ZBD1_PFLI16-CARRID
        AND CONNID = ZBD1_PFLI16-CONNID.


      IF SY-SUBRC = 0.
        MESSAGE S000(OO) WITH '이미 존재하는 데이터입니다.' DISPLAY LIKE 'E'.
        EXIT.
      ELSE.
        PERFORM MODIFY_TABLE.
      ENDIF.

    WHEN 'B'.  " 수정
      PERFORM MODIFY_TABLE.
  ENDCASE.

  CLEAR ZBD1_PFLI16.

ENDFORM.

      OTHERS   = 2.
ENDFORM.

*& Form CHANGE_DATA
*&---------------------------------------------------------------------*
FORM CHANGE_DATA .
  CALL METHOD CL_ALV_GRID->GET_SELECTED_ROWS
    IMPORTING
      ET_ROW_NO = LT_ROW_NO.            " Numeric IDs of Selected Rows

  IF LT_ROW_NO IS INITIAL.
    MESSAGE E000(OO) WITH '수정할 줄을 선택해주세요.'.
  ENDIF.

  READ TABLE LT_ROW_NO INDEX 1 INTO LS_ROW_NO.

  READ TABLE GT_SPFLI INDEX LS_ROW_NO-ROW_ID INTO ZBD1_PFLI16.

ENDFORM.

FORM MODIFY_TABLE .
  MODIFY ZBD1_PFLI16 FROM ZBD1_PFLI16.
  MESSAGE S000(OO) WITH '데이터가 성공적으로 저장됐습니다.'.
