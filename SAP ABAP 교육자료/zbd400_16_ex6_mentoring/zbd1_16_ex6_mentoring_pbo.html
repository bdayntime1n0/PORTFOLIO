MODULE MODIFY_SCREEN_0200 OUTPUT.
  CASE GV_MODE. " A = 생성, B - 수정, C = 삭제
    WHEN 'A'.

    WHEN 'B'.
      LOOP AT SCREEN.
        CASE SCREEN-GROUP1.
          WHEN 'GR1'.
            SCREEN-INPUT = 0.
          WHEN OTHERS.
            SCREEN-INPUT = 1.
        ENDCASE.
        MODIFY SCREEN.
      ENDLOOP.

    WHEN 'C'.

    WHEN 'D'.
      LOOP AT SCREEN.
        CASE SCREEN-INPUT.
          WHEN '1'.
            SCREEN-INPUT = 0.
        ENDCASE.
        MODIFY SCREEN.
      ENDLOOP.
  ENDCASE.
ENDMODULE.
