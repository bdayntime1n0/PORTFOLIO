REPORT ZBD400_16_IF_2.


PARAMETERS P_CARRID TYPE S_CARR_ID.
PARAMETERS P_FLDATE TYPE S_FLDATE1.

IF ( P_CARRID = 'AA' OR P_CARRID = 'LH' ) AND
         P_FLDATE = SY-DATUM.

  WRITE 'P_CARRID 는 AA 이거나 LH 이면서 P_FLDATE 가 오늘입니다.'.

ELSEIF ( P_CARRID = 'UA' OR P_CARRID = 'DL' ) AND
         P_FLDATE > SY-DATUM.

  WRITE 'P_CARRID 는 UA 이거나 DL 이면서 P_FLDATE 가 오늘 이후입니다.'.

ENDIF.
