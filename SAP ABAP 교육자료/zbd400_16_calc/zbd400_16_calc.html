REPORT ZBD400_16_CALC.


* 전역변수 선언
DATA : GV_OCC TYPE SBC400FOCC-SEATSOCC,
       GV_PERCENTAGE  TYPE SBC400FOCC-PERCENTAGE.

* PARAMETER 선언. 8자리 제한.
PARAMETERS P_MAX TYPE SBC400FOCC-SEATSMAX.

* 계산.
IF P_MAX = 0.
*  P_MAX = 0 일 때.

WRITE : /1 'P_MAX 는 0 입니다.'.

*  P_MAX = 0 이 아닐 때.

GV_PERCENTAGE = GV_OCC * 100 / P_MAX.
WRITE : /1 '계산 결과는 ', GV_PERCENTAGE, ' 입니다.'.

WRITE /1 '계산 결과는 '.
WRITE GV_PERCENTAGE.
WRITE ' 입니다.'.

ENDIF.
