*&---------------------------------------------------------------------*
FORM SELECTED_DATA_RTN .

  REFRESH GT_DISPLAY.

  SELECT C~CARRID,
         A~CARRNAME,
         B~CONNID,
         C~FLDATE,
         B~COUNTRYFR,
         B~COUNTRYTO,
         C~SEATSMAX_F,  " 최대좌석 First
         C~SEATSMAX_B,  " 최대좌석 Business
         C~SEATSMAX,    " 최대좌석 Economy
         C~SEATSOCC_F,  " 현재예약된 First
         C~SEATSOCC_B,  " 현재예약된 Business
         C~SEATSOCC,    " 현재예약된 Economy
         C~PRICE,       " 운임
         C~CURRENCY     " 통화
    FROM SCARR AS A INNER JOIN SPFLI   AS B  ON B~CARRID EQ A~CARRID
                    INNER JOIN SFLIGHT AS C  ON C~CARRID EQ B~CARRID
                                            AND C~CONNID EQ B~CONNID

   " Selection Screen 에서 검색조건들을 누락없이 적용한다.
   WHERE C~CARRID IN @S_CARRID
     AND C~CONNID IN @S_CONNID
     AND C~FLDATE IN @S_FLDATE

    INTO TABLE @GT_DISPLAY.

  IF SY-SUBRC EQ 0.
    SORT GT_DISPLAY BY CARRID
                       CONNID
                       FLDATE.
  ENDIF.

ENDFORM.

FORM GET_FIELDCATALOG_BY_FUNC.

* DATA : GT_FIELDCAT TYPE LVC_T_FCAT.

*DATA : LT_FIELDCAT TYPE SLIS_T_FIELDCAT_ALV. " 이거도 있음. 근데 많이 구식.

  CALL FUNCTION 'K_KKB_FIELDCAT_MERGE'
    EXPORTING
      I_CALLBACK_PROGRAM     = SY-REPID       " Internal table declaration program
      I_INCLNAME             = SY-REPID
      I_TABNAME              = 'GS_DISPLAY'   " Name of table to be displayed
    CHANGING
      CT_FIELDCAT            = LT_FIELDCAT    " Field Catalog with Field Descriptions
    EXCEPTIONS
      INCONSISTENT_INTERFACE = 1
      OTHERS                 = 2.

  CALL FUNCTION 'LVC_TRANSFER_FROM_KKBLO'     " KKBLO로 만들어진 TABLE을 LVC로 받을 수 있음.
    EXPORTING        " <==
      IT_FIELDCAT_KKBLO = LT_FIELDCAT   " IT_FIELDCAT으로 준다
    IMPORTING        " ==>
      ET_FIELDCAT_LVC   = GT_FIELDCAT   " ET_FIELDCAT으로부터 가져옴
    EXCEPTIONS
      IT_DATA_MISSING   = 1
      OTHERS            = 2.

ENDFORM.
*&---------------------------------------------------------------------*
*& Form MAKE_FIELDCATALOG
*&---------------------------------------------------------------------*
FORM MAKE_FIELDCATALOG .
  " 필드마다 색을 줄 수 있음 -> 텍스트는 초록색, 금액은 노란색으로 많이 줌.

  DATA : LV_TEXT TYPE TEXT100.

  LOOP AT GT_FIELDCAT INTO GS_FIELDCAT.

    CLEAR LV_TEXT.

    CASE GS_FIELDCAT-FIELDNAME.
        " 각 필드별 옵션 설정을 함.
        " field name이 carrid일 때만 text를 항공사id로 바꾸고 나머지는 지나감
      WHEN 'CARRID'.  " field name이 carrid일 때만 text를 항공사id로 바꾸고 나머지는 지나감
        LV_TEXT = '항공사ID'.
        GS_FIELDCAT-JUST = 'C'.      " 정렬. C = 가운데 정렬, L = 왼쪽, R = 오른쪽
      WHEN 'CARRNAME'.
        GS_FIELDCAT-EMPHASIZE = 'C500'.  " 색상은 4자리. C500은 연초록
      WHEN 'CNTRFR'.
        LV_TEXT = '출발국가'.
      WHEN 'CNTRTO'.
        LV_TEXT = '도착국가'.
      WHEN 'PRICE'.
        GS_FIELDCAT-EMPHASIZE = 'C300'.  " C300 = 연노랑

      WHEN OTHERS.
    ENDCASE.

    IF LV_TEXT IS NOT INITIAL.
      " 여기 구문이 실행되는 경우는 위의 CASE에서 특정 필드가 LV_TEXT의 값을 지정한 경우에 해당
      GS_FIELDCAT-SCRTEXT_L = LV_TEXT.
      GS_FIELDCAT-SCRTEXT_M = LV_TEXT.
      GS_FIELDCAT-SCRTEXT_S = LV_TEXT.
      GS_FIELDCAT-COLTEXT   = LV_TEXT.
      GS_FIELDCAT-REPTEXT   = LV_TEXT.
    ENDIF.
    " INTERNAL TABLE 에 GS_FIELDCAT 의 내용을 반영.
    MODIFY GT_FIELDCAT FROM GS_FIELDCAT.

  ENDLOOP.

ENDFORM.
