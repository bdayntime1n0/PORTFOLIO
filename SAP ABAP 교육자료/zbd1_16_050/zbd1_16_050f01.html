*&---------------------------------------------------------------------*
FORM SELECTED_DATA_RTN .

  REFRESH GT_DISPLAY.

  SELECT C~CARRID,
         A~CARRNAME,
         B~CONNID,
         C~FLDATE,
         B~COUNTRYFR,
         B~COUNTRYTO,
         C~SEATSMAX_F,  " 최대좌석 First
         C~SEATSMAX_B,  " 최대좌석 Business
         C~SEATSMAX,    " 최대좌석 Economy
         C~SEATSOCC_F,  " 현재예약된 First
         C~SEATSOCC_B,  " 현재예약된 Business
         C~SEATSOCC,    " 현재예약된 Economy
         C~PRICE,       " 운임
         C~CURRENCY     " 통화
    FROM SCARR AS A INNER JOIN SPFLI   AS B  ON B~CARRID EQ A~CARRID
                    INNER JOIN SFLIGHT AS C  ON C~CARRID EQ B~CARRID
                                            AND C~CONNID EQ B~CONNID

   " Selection Screen 에서 검색조건들을 누락없이 적용한다.
   WHERE C~CARRID IN @S_CARRID
     AND C~CONNID IN @S_CONNID
     AND C~FLDATE IN @S_FLDATE

    INTO TABLE @GT_DISPLAY.

  IF SY-SUBRC EQ 0.
    SORT GT_DISPLAY BY CARRID
                       CONNID
                       FLDATE.
  ENDIF.

ENDFORM.

FORM MAKE_FIELDCATALOG .

  DATA LV_TEXT TYPE TEXT100.

  LOOP AT GT_FIELDCAT INTO GS_FIELDCAT. " 첫번째 필드부터 마지막 필드까지

    CLEAR LV_TEXT.

    CASE GS_FIELDCAT-FIELDNAME.
*     WHEN '필드이름'. 각 필드별 옵션 설정을 한다.
      WHEN 'CARRID'.
        LV_TEXT = '항공사ID'.
        GS_FIELDCAT-JUST = 'C'. " C : Center, L, R
      WHEN 'CNTRFR'.
        LV_TEXT = '출발국가'.
      WHEN 'CNTRTO'.
        LV_TEXT = '도착국가'.
      WHEN 'CARRNAME'.
        GS_FIELDCAT-EMPHASIZE = 'C500'. " 색상에 대한 정보를 다루는 필드
      WHEN 'PRICE'.
        GS_FIELDCAT-EMPHASIZE = 'C300'.
      WHEN OTHERS.
    ENDCASE.


    IF LV_TEXT IS NOT INITIAL.
      " 여기 구문이 실행되는 경우는
      " 위의 CASE 에서 특정 필드가 LV_TEXT 에 값을 지정한 경우에 해당한다.
      GS_FIELDCAT-SCRTEXT_L = LV_TEXT.
      GS_FIELDCAT-SCRTEXT_M = LV_TEXT.
      GS_FIELDCAT-SCRTEXT_S = LV_TEXT.
      GS_FIELDCAT-COLTEXT   = LV_TEXT.
      GS_FIELDCAT-REPTEXT   = LV_TEXT.
    ENDIF.


    " Internal Table 에 GS_FIELDCAT 의 내용을 반영한다.
    MODIFY GT_FIELDCAT FROM GS_FIELDCAT.

  ENDLOOP.

ENDFORM.

FORM EVENT_TOOLBAR

  USING PO_OBJECT TYPE REF TO CL_ALV_EVENT_TOOLBAR_SET
        PO_SENDER TYPE REF TO CL_GUI_ALV_GRID.

  " PO_SENDER 가 GO_GRID 와 동일한 ALV 를 가리킬 때만 진행한다.
  CHECK PO_SENDER EQ GO_GRID.

  DATA LT_TOOLBAR TYPE TTB_BUTTON.
  DATA LS_TOOLBAR TYPE STB_BUTTON.

  LT_TOOLBAR = PO_OBJECT->MT_TOOLBAR. " MT => Member Internal Table

  " 버튼과 버튼사이를 구분하는 구분자 추가
  CLEAR LS_TOOLBAR.
  LS_TOOLBAR-BUTN_TYPE = 3.
  APPEND LS_TOOLBAR TO LT_TOOLBAR.

  " 현재 라인수를 보여주는 버튼 추가
  DATA LV_LINES TYPE I.
  LV_LINES = LINES( GT_DISPLAY ). " Internal Table 의 현재 라인을 가져오는 함수

  CLEAR LS_TOOLBAR.
  LS_TOOLBAR-BUTN_TYPE = 0.
  LS_TOOLBAR-TEXT      = 'Lines:' && LV_LINES.
  APPEND LS_TOOLBAR TO LT_TOOLBAR.

  " 1. 현재 선택한 라인의 예약된 좌석 수를 1 증가하는 버튼 추가.
  CLEAR LS_TOOLBAR.
  LS_TOOLBAR-BUTN_TYPE = 0.
  LS_TOOLBAR-FUNCTION  = 'OCC_ADD'.            " 버튼에 function code를 준다고 생각하면 됨.
  LS_TOOLBAR-ICON      = ICON_POSITIVE.
  LS_TOOLBAR-TEXT      = '예약증가'.
  APPEND LS_TOOLBAR TO LT_TOOLBAR.

  " 2. 현재 선택한 라인의 예약된 좌석 수를 1 감소하는 버튼 추가.
  CLEAR LS_TOOLBAR.
  LS_TOOLBAR-BUTN_TYPE = 0.
  LS_TOOLBAR-FUNCTION  = 'OCC_SUB'.
  LS_TOOLBAR-ICON      = ICON_NEGATIVE.
  LS_TOOLBAR-TEXT      = '예약감소'.
  APPEND LS_TOOLBAR TO LT_TOOLBAR.

  PO_OBJECT->MT_TOOLBAR = LT_TOOLBAR.

ENDFORM.

FORM EVENT_USER_COMMAND
  USING    PV_UCOMM  TYPE SY-UCOMM    " 일반변수이기 때문에 V. PARAMETER VARIABLE.
           PO_SENDER TYPE REF TO CL_GUI_ALV_GRID.

  CHECK PO_SENDER EQ GO_GRID.

  DATA LT_ROW TYPE LVC_T_ROW. " 현재 선택한 라인 정보를 가질 INTERNAL TABLE.
  DATA LS_ROW TYPE LVC_S_ROW.
  DATA LV_REFRESH TYPE C.     " ALV의 새로고침 여부를 결정하는 FLAG 변수.

  CALL METHOD PO_SENDER->GET_SELECTED_ROWS
    IMPORTING
      ET_INDEX_ROWS = LT_ROW                " Indexes of Selected Rows
*     ET_ROW_NO     =                  " Numeric IDs of Selected Rows
    .

  CASE PV_UCOMM.
    WHEN 'OCC_ADD'.
      " 현재 라인의 예약을 증가시킴

      LOOP AT LT_ROW INTO LS_ROW.

        READ TABLE GT_DISPLAY INTO GS_DISPLAY INDEX LS_ROW-INDEX.

        IF SY-SUBRC = 0.
          IF GS_DISPLAY-SEATSMAX > GS_DISPLAY-SEATSOCC.
            ADD 1 TO GS_DISPLAY-SEATSOCC.
            MODIFY GT_DISPLAY FROM GS_DISPLAY        " GS_DISPLAY 에서 가져와 수정해라.
                              INDEX LS_ROW-INDEX     " 선택한 라인의
                              TRANSPORTING SEATSOCC. " SEATSOCC 필드를

            IF SY-SUBRC EQ 0.
              " ITAB의 내용이 변경되면 변경된 데이터를 보여주기 위해 새고로침이 필요함.
              " 새로고침 실행여부를 결정짓는 플래그 변수에 'X' 값을 기록.
              LV_REFRESH = 'X'.
            ENDIF.
          ELSE.
            " ERROR -> 예약이 가득 찬 비행기이기 때문에 증가 안됨.
          ENDIF.
        ENDIF.
      ENDLOOP.

    WHEN 'OCC_SUB'.
      " 현재 라인의 예약을 감소시킴
      LOOP AT LT_ROW INTO LS_ROW.

        READ TABLE GT_DISPLAY INTO GS_DISPLAY INDEX LS_ROW-INDEX.

        IF SY-SUBRC = 0.
          IF GS_DISPLAY-SEATSOCC > 0.
            SUBTRACT 1 FROM GS_DISPLAY-SEATSOCC.
            MODIFY GT_DISPLAY FROM GS_DISPLAY        " GS_DISPLAY 에서 가져와 수정해라.
                              INDEX LS_ROW-INDEX     " 선택한 라인의
                              TRANSPORTING SEATSOCC. " SEATSOCC 필드를

            IF SY-SUBRC EQ 0.
              " ITAB의 내용이 변경되면 변경된 데이터를 보여주기 위해 새고로침이 필요함.
              " 새로고침 실행여부를 결정짓는 플래그 변수에 'X' 값을 기록.
              LV_REFRESH = 'X'.
            ENDIF.
          ELSE.
            " ERROR -> 예약이 가득 찬 비행기이기 때문에 증가 안됨.
          ENDIF.
        ENDIF.
      ENDLOOP.
  ENDCASE.

  IF LV_REFRESH EQ 'X'.

    DATA LS_STABLE TYPE LVC_S_STBL.  " ROW, COLUMN FIELD 있음.
    " ROW 에 X를 주면 행 선택 상태 유지.
    " COLUMN 에 X를 주면 열 선택 상태 유지.
    LS_STABLE-ROW = 'X'.
    LS_STABLE-COL = 'X'.

    CALL METHOD PO_SENDER->REFRESH_TABLE_DISPLAY
      EXPORTING
        IS_STABLE = LS_STABLE   " With Stable Rows/Columns
*       I_SOFT_REFRESH =             " Without Sort, Filter, etc.
      EXCEPTIONS
        FINISHED  = 1                " Display was Ended (by Export)
        OTHERS    = 2.

    " LT_ROW = 선택했던 라인들에 대한 정보를 갖고 있음.

    " 새로고침 이후 LT_ROW를 이용해서 기존 선택했던 라인을 재선택하게 함.
    CALL METHOD PO_SENDER->SET_SELECTED_ROWS

        IT_INDEX_ROWS = LT_ROW  " Indexes of Selected Rows
*       IT_ROW_NO     =        " Numeric Row IDs
*       IS_KEEP_OTHER_SELECTIONS =        " Character Field of Length 1
      .
  ENDIF.
