REPORT ZBD400_16_ITAB_SELECT_SINGLE_2.


PARAMETERS : P_CARRID TYPE SFLIGHT-CARRID,
             P_CONNID TYPE SFLIGHT-CONNID,
             P_FLDATE TYPE SFLIGHT-FLDATE.


START-OF-SELECTION.

  DATA : LS_FLIGHT TYPE BC400_S_FLIGHT.

  SELECT SINGLE SEATSMAX SEATSOCC
    FROM SFLIGHT
    INTO CORRESPONDING FIELDS OF LS_FLIGHT
    WHERE CARRID = P_CARRID
      AND CONNID = P_CONNID
      AND FLDATE = P_FLDATE.

  IF SY-SUBRC = 0.
    WRITE : / LS_FLIGHT-CARRID,
              LS_FLIGHT-CONNID,
              LS_FLIGHT-FLDATE,
              LS_FLIGHT-SEATSMAX,
              LS_FLIGHT-SEATSOCC.

  ELSE.
    WRITE : / 'NO DATA FOUND.'.
  ENDIF.
