CLASS LCL_AIRPLANE IMPLEMENTATION.
  METHOD CLASS_CONSTRUCTOR.
    SELECT *
      FROM SAPLANE
      INTO CORRESPONDING FIELDS OF TABLE GT_PLANETYPES.
  ENDMETHOD.


  METHOD CONSTRUCTOR.
    MV_NAME      = IV_NAME.
    MV_PLANETYPE = IV_PLANETYPE.
    ADD 1 TO GV_N_O_AIRPLANES.
    " CREATE OBJECT 시 자동으로 생성되는 생성자의 위치에 EVENT 로직을 입력하는 것이 좋음.
    RAISE EVENT AIRPLANE_CREATED.
  ENDMETHOD.

*  METHOD SET_ATTRIBUTES.
*    MV_NAME      = IV_NAME.
*    MV_PLANETYPE = IV_PLANETYPE.
*    ADD 1 TO GV_N_O_AIRPLANES.
*  ENDMETHOD.

  METHOD DISPLAY_ATTRIBUTES.
    WRITE : /(10) 'NAME : ',
             (20) MV_NAME,
             (10) 'TYPE : ',
             (20) MV_PLANETYPE.
  ENDMETHOD.

  METHOD DISPLAY_N_O_AIRPLANES.
    WRITE :/ 'AIRPLANE COUNT : ', GV_N_O_AIRPLANES.
  ENDMETHOD.
ENDCLASS.
