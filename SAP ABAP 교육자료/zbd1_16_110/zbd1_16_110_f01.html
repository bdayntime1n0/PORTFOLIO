FORM GET_DATA_FROM_EXCEL .

  " Top Include 에 전역변수로 선언해둬야 한다.
*  DATA GT_INTERN TYPE TABLE OF ALSMEX_TABLINE.

  REFRESH GT_INTERN.

  " 5000개씩 가져올 임시 테이블로 활용
  DATA LT_INTERN TYPE TABLE OF ALSMEX_TABLINE.

  DATA LV_BEGIN_ROW TYPE I.
  DATA LV_END_ROW   TYPE I.
  DATA LV_LINES     TYPE I.


  " 조회를 시작할 행번호와 한번에 최대 5000줄를 가져오기 위한 행번호 설정
  CONSTANTS LC_MAX_LINE TYPE I VALUE 5000.

  " ex) 시작행번호가 1일 경우 4999 를 더하면 5000 번째 줄까지 가져온다.
  " ex) 시작행번호가 2일 경우 4999 를 더하면 5001 번째 줄까지 가져온다.
  LV_BEGIN_ROW = 2.
  LV_END_ROW   = LV_BEGIN_ROW + LC_MAX_LINE - 1.

  DO.

    REFRESH LT_INTERN.

    " 엑셀 파일에서 내용을 읽어와 Internal Table 에 보관
    CALL FUNCTION 'ALSM_EXCEL_TO_INTERNAL_TABLE'
      EXPORTING
        FILENAME                = P_FILE
        I_BEGIN_COL             = 1
        I_BEGIN_ROW             = LV_BEGIN_ROW    " 데이터 조회를 시작할 행번호
        I_END_COL               = 10
        I_END_ROW               = LV_END_ROW      " 데이터 조회를 종료할 행번호
      TABLES
        INTERN                  = LT_INTERN
      EXCEPTIONS
        INCONSISTENT_PARAMETERS = 1
        UPLOAD_OLE              = 2
        OTHERS                  = 3.

    " 엑셀에서 가져온 데이터의 라인 정보를 보관
    CLEAR LV_LINES.
    SORT LT_INTERN BY ROW COL.
    LOOP AT LT_INTERN INTO DATA(LS_INTERN).
      AT NEW ROW.   " ROW 필드의 값이 변경될 떄 아래 구문을 실행
        ADD 1 TO LV_LINES.
      ENDAT.
    ENDLOOP.

    IF LV_LINES EQ LC_MAX_LINE.

      " 가져온 데이터를 전역변수에 보관
      APPEND LINES OF LT_INTERN TO GT_INTERN.

      " 조회가능한 최대 라인수까지 가져왔을 때는
      " 데이터가 더 존재할 수 있으므로,
      " 조회시작행번호와 조회종료행번호를 다음 단계로 조정한다.

      LV_BEGIN_ROW = LV_END_ROW + 1.            " 방금 조회했던 라인의 다음줄부터
      LV_END_ROW   = LV_END_ROW + LC_MAX_LINE.  " 최대 라인수만큼 더 조회를 한다.

    ELSE.

      IF LV_LINES > 0.
        " 가져온 데이터를 전역변수에 보관
        APPEND LINES OF LT_INTERN TO GT_INTERN.
      ENDIF.

      EXIT.

    ENDIF.

  ENDDO.

ENDFORM.

*& Form MOVE_INTERN_TO_ITAB

FORM MOVE_INTERN_TO_ITAB .

  " GT_INTERN 을 Internal Table 에 올바르게 매칭시켜 데이터를 옮겨 담는 로직

  FIELD-SYMBOLS <FS>.

  " 행->열 순으로 정렬한 뒤에 진행
  SORT GT_INTERN BY ROW COL.

  LOOP AT GT_INTERN INTO DATA(LS_INTERN).
    AT NEW ROW.
      CLEAR GS_DISPLAY.
      GS_DISPLAY-STATUS = ICON_YELLOW_LIGHT. " 신호등 표현을 위한 ICON.
    ENDAT.

    " STATUS 아이콘을 넣기 위해 COL을 한줄씩 밀어줘야 함.
    LS_INTERN-COL = LS_INTERN-COL + 1.

    ASSIGN COMPONENT LS_INTERN-COL OF STRUCTURE GS_DISPLAY TO <FS>.
    IF SY-SUBRC EQ 0.
      CASE LS_INTERN-COL.
        WHEN 3.
          " REPLACE 라는 키워드는 IN 뒤에 나오는 텍스트에 대해서
          " 특정 단어를 검색해서 다른 단어로 변경해주는 기능을 한다.
          " 아래와 같은 경우 LS_INTERN-VALUE 에 '2021-01-01' 이라는
          " 텍스트가 존재할 때 '-' 를 찾아서 '' 으로 변경해준다.
          " 그리하여 LS_INTERN-VALUE 의 내용이 '2021-01-01' 에서 '20210101' 로 변경된다.
          REPLACE ALL OCCURRENCES OF '-' IN LS_INTERN-VALUE WITH ''.
      ENDCASE.

      <FS> = LS_INTERN-VALUE.
      UNASSIGN <FS>.
    ENDIF.

    AT END OF ROW.
      APPEND GS_DISPLAY TO GT_DISPLAY.
    ENDAT.
  ENDLOOP.


ENDFORM.

FORM SAVE_DATA .
  " INTERNAL TABLE에 들어있는 내용을
  " T-CODE KS01에 반복적으로 수행하여
  " COST CENTER를 생성하는 것이 목적.

  " BDC 녹음 뜬 것을 그대로 옮기기 위해 BDC 모양의 ITAB 생성.
  DATA LT_BDC TYPE TABLE OF BDCDATA.
  DATA LS_BDC TYPE BDCDATA.

  DATA LS_OPTION TYPE CTU_PARAMS.

  DATA LT_MESSAGE TYPE TABLE OF BDCMSGCOLL.

  " 매크로를 사용해서 COST CENTER 자동생성.
  DEFINE __ADD_BDC_FIELD.
    CLEAR LS_BDC.

  LS_BDC-FNAM = &1.
  LS_BDC-FVAL = &2.
  APPEND LS_BDC TO LT_BDC.
  END-OF-DEFINITION.

  LOOP AT GT_DISPLAY INTO GS_DISPLAY.

    REFRESH LT_BDC.

    CLEAR LS_BDC.
    LS_BDC-PROGRAM  = 'SAPLKMA1'.
    LS_BDC-DYNPRO   = '0200'.
    LS_BDC-DYNBEGIN = 'X'.
    APPEND LS_BDC TO LT_BDC.

    " 화면에 등장하면 필드에 대한 값지정
    " 값 지정 이후 실행할 FUNCTION 지정
    __ADD_BDC_FIELD : 'BDC_OKCODE'        '/00',
                      'CSKSZ-KOKRS'       '0001',
                      'CSKSZ-KOSTL'       GS_DISPLAY-KOSTL,
                      'CSKSZ-DATAB_ANFO'  GS_DISPLAY-DATAB,
                      'CSKSZ-DATBI_ANFO'  GS_DISPLAY-DATBI.

    " 화면에 대한 내용
    CLEAR LS_BDC.
    LS_BDC-PROGRAM  = 'SAPLKMA1'.
    LS_BDC-DYNPRO   = '0299'.
    LS_BDC-DYNBEGIN = 'X'.
    APPEND LS_BDC TO LT_BDC.

    " 화면에 등장하면 필드에 대한 값지정
    " 값 지정 이후 실행할 FUNCTION 지정
    __ADD_BDC_FIELD : 'BDC_OKCODE'   '=BU',   " 저장버튼
                      'CSKSZ-KTEXT'  GS_DISPLAY-KTEXT,
                      'CSKSZ-LTEXT'  GS_DISPLAY-LTEXT,
                      'CSKSZ-VERAK'  GS_DISPLAY-VERAK,
                      'CSKSZ-KOSAR'  GS_DISPLAY-KOSAR,
                      'CSKSZ-KHINR'  GS_DISPLAY-KHINR,
                      'CSKSZ-WAERS'  GS_DISPLAY-WAERS,
                      'CSKSZ-PRCTR'  GS_DISPLAY-PRCTR.



    " BDC 실행옵션
    LS_OPTION-DISMODE = 'N'.
    " A : LT_BDC 의 모든 화면을 전부 사용자에게 보여주면서 진행.
    " E : LT_BDC 의 모든 화면을 백그라운드로 실행, 오류 시 사용자에게 보여줌. ( 테스트용)
    " N : LT_BDC 의 모든 화면을 백그라운드로 실행, 사용자에게 화면 제시 X. ( 실사용 )

    LS_OPTION-UPDMODE = 'S'.  " S : 동기화 모드
                              " A : 비동기화 모드

    CLEAR LT_MESSAGE.
    " KS01에 들어가서 코스트센터 생성.
    CALL TRANSACTION 'KS01' USING LT_BDC              " BDC ITAB
                            OPTIONS FROM LS_OPTION    " 추가 옵션
                            MESSAGES INTO LT_MESSAGE. " 오류 메시지 출력을 위함

    " BDC 실행결과를 점검하는 구간.
