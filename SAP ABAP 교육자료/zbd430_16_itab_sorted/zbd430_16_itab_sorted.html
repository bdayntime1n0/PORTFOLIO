DATA it_flight TYPE ZBD1_T_SFLIGHT16.
DATA wa_sflight TYPE sflight.


WRITE / 'Printout in tableorder of Database:'.

SELECT * FROM sflight
 INTO wa_sflight
 WHERE carrid = 'JL'.
  WRITE: / wa_sflight-carrid,
           wa_sflight-connid,
           wa_sflight-fldate,
           wa_sflight-price,
           wa_sflight-currency,
           wa_sflight-planetype.
ENDSELECT. " ENDSELECT = 한 줄 씩 반복해서 출력.

ULINE. " 밑줄

SELECT * FROM sflight
 INTO TABLE it_flight               " ARRAY FETCH 방식으로 INTERNAL TABLE에 저장.
 WHERE carrid = 'JL'.               " SORTED TABLE. FLDATE를 기준으로 정렬됨.
WRITE / 'Printout in tableorder of sorted ITAB:'.

LOOP AT it_flight INTO wa_sflight.
  WRITE: / wa_sflight-carrid,
           wa_sflight-connid,
           wa_sflight-fldate,
           wa_sflight-price,
