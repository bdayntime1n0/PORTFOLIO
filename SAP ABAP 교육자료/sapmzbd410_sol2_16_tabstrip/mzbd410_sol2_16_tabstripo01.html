MODULE MODIFY_SCREEN_0100 OUTPUT.

  CASE 'X'.
    WHEN VIEW.
    WHEN MAINTAIN_FLIGHTS.
      LOOP AT SCREEN.
        CASE SCREEN-NAME.
          WHEN 'SDYN_CONN-PLANETYPE'.
            SCREEN-INPUT = 1.
        ENDCASE.
        MODIFY SCREEN.

      ENDLOOP.
    WHEN MAINTAIN_BOOKINGS.

  ENDCASE.

ENDMODULE.

MODULE GET_SPFLI OUTPUT.

  CLEAR WA_SPFLI.

  SELECT SINGLE *
    FROM ZBD1_PFLI16
    INTO WA_SPFLI
    WHERE CARRID = SDYN_CONN-CARRID
      AND CONNID = SDYN_CONN-CONNID.

  " 화면에 데이터를 출력하기 위해 MOVE-CORRESPONDING 함.
  MOVE-CORRESPONDING WA_SPFLI TO SDYN_CONN.

ENDMODULE.
