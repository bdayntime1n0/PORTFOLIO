FORM LINE_CHECK TABLES PT_ROW TYPE LVC_T_ROW USING PV_VAL PV_LINE.

  DATA : LS_ROW TYPE LVC_S_ROW.
  DATA : LV_MSG(50).

  CASE PV_VAL.

    WHEN 'C'.
      LV_MSG = '## # ## ### ######.'.

  ENDCASE.

  IF PT_ROW[] IS INITIAL.
    MESSAGE I000(OO) WITH '### ### ####.'.

  ELSEIF PV_LINE GT 1 AND LV_MSG IS NOT INITIAL.
    MESSAGE I000(OO) WITH LV_MSG.

  ELSE.

    LOOP AT PT_ROW INTO LS_ROW.

      READ TABLE GT_ORDER_DATA INTO DATA(LS_DATA) INDEX LS_ROW-INDEX.

      GR_ORNM-OPTION = 'EQ'.
      GR_ORNM-SIGN   = 'I'.
      GR_ORNM-LOW    = LS_DATA-ZORDERNUM.

      CASE PV_VAL.

        WHEN 'C'.

          APPEND GR_ORNM.

          PERFORM STATUS_DETAIL TABLES GR_ORNM.

          PERFORM ORDER_CANCEL USING LS_DATA-ZORDERNUM.


        WHEN 'DT'.

          APPEND GR_ORNM.

          PERFORM STATUS_DETAIL TABLES GR_ORNM.

          PERFORM SAPGUI_PROGRESS_INDICATOR USING TEXT-M07.

        WHEN 'RE'.

          APPEND GR_ORNM.

          PERFORM STATUS_DETAIL TABLES GR_ORNM.

        WHEN 'DL'.

          DELETE GT_STATUS_DATA INDEX LS_ROW-INDEX.

        WHEN 'B'.

*          PERFORM BILLING_BAPI.

      ENDCASE.

      PERFORM REFRESH_GRID  USING GO_GRID2.

    ENDLOOP.

    CLEAR : GR_ORNM[], GR_ORNM.

  ENDIF.

ENDFORM.
