FORM MOVE_EXCEL_ITAB .

  DATA : LV_LINE TYPE SY-DBCNT.

  " #### ## ##
  FIELD-SYMBOLS <FS>.

  LOOP AT GT_EXCEL INTO DATA(LS_EXCEL).

    AT NEW ROW.
      CLEAR GS_DATA.
      GS_DATA-STATUS = ICON_YELLOW_LIGHT.
    ENDAT.

    LS_EXCEL-COL = LS_EXCEL-COL + 2.

    ASSIGN COMPONENT LS_EXCEL-COL OF STRUCTURE GS_DATA TO <FS>.

    " # ### #### ## ##
    IF SY-SUBRC EQ 0.
      CASE LS_EXCEL-COL.
        WHEN 21.
          REPLACE ALL OCCURRENCES OF ',' IN LS_EXCEL-VALUE WITH ''.
        WHEN 23.
          REPLACE ALL OCCURRENCES OF ',' IN LS_EXCEL-VALUE WITH ''.
        WHEN 25.
          REPLACE ALL OCCURRENCES OF '-' IN LS_EXCEL-VALUE WITH ''.
      ENDCASE.

      <FS> = LS_EXCEL-VALUE.

      UNASSIGN <FS>.

    ENDIF.

    AT END OF ROW.

      PERFORM AMT_CONV_TOEXTERNAL USING GS_DATA-ZAMOUNT_CUKY CHANGING GS_DATA-ZAMOUNT.

      PERFORM DUPLICATE_CODE  USING GS_DATA-ZUNIPASS  CHANGING GS_DATA-ZCUSTOMID
                                    GS_DATA-ZVENDORNM          GS_DATA-ZVENDORCO
                                    GS_DATA-ZPRODNM            GS_DATA-ZPRODCO.

      PERFORM NEW_CODE USING GS_DATA-ZORDERDT CHANGING GS_DATA-ZORDERNUM
                                                       GS_DATA-ZCUSTOMID
                                                       GS_DATA-ZVENDORCO
                                                       GS_DATA-ZPRODCO.

      APPEND GS_DATA TO GT_CPDT.

      CLEAR GS_DATA.

    ENDAT.

    " # ### #### ## ##
    PERFORM SAVE_EXCEL.

    MOVE-CORRESPONDING GT_CPDT[] TO GT_DATA[].

  ENDLOOP.

  LV_LINE = LINES( GT_DATA ).

  MESSAGE S000(OO) WITH LV_LINE && ' # #######.'.

  CLEAR : GT_CPDT[], GT_CPDT.

ENDFORM.

FORM DUPLICATE_CODE USING PV_ZUNIPASS CHANGING PV_ZCUSTOMID
                          PV_ZVENDORNM           PV_ZVENDORCO
                          PV_ZPRODNM             PV_ZPRODCO.

  CLEAR : PV_ZCUSTOMID,
          PV_ZVENDORCO,
          PV_ZPRODCO.

  SELECT SINGLE ZCUSTOMID
    FROM ZTOI_CONSUMER
    INTO PV_ZCUSTOMID
   WHERE ZUNIPASS EQ PV_ZUNIPASS.

  IF SY-SUBRC NE 0.
    PV_ZCUSTOMID = ''.
  ENDIF.

  SELECT SINGLE ZVENDORCO
    FROM ZTOI_VENDOR
    INTO PV_ZVENDORCO
   WHERE ZVENDORNM EQ PV_ZVENDORNM.

  IF SY-SUBRC NE 0.
    PV_ZVENDORCO = ''.
  ENDIF.

  SELECT SINGLE ZPRODCO
    FROM ZTOI_PRODUCT
    INTO PV_ZPRODCO
   WHERE ZPRODNM EQ PV_ZPRODNM.

  IF SY-SUBRC NE 0.
    PV_ZPRODCO = ''.
  ENDIF.

ENDFORM.
