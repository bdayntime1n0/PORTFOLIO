FORM GET_DATA.

  SELECT
    T~YEARMONTH,
    O~ZORDERNUM,
*    S~ZRETCHK,
*    D~ZDELIVST,
    S~ZSOFLAG,
    O~ZCUSTOMID,
    C~ZCUSTOMNM,
    C~ZPHONENM,
    O~ZPRODCO,
    P~ZCATEGORY,
    TX~DDTEXT AS ZCATEGORY_TX,
    P~ZPRODNM,
    O~ZORDERDT,
*    O~ZDELVDT,
    O~ZUNIT,
    O~ZAMOUNT_CUKY,
    O~ZQUANTITY,  "##
    O~ZAMOUNT,  "####
*    O~ZORDERNUM,
*    O~ZPRODCO,
*    P~ZCATEGORY,
*    T~YEARMONTH,
*    O~ZUNIT,
*    O~ZAMOUNT_CUKY,
*    O~ZORDERDT,
*    O~ZQUANTITY,  "##
*    O~ZAMOUNT  "####
      p~ZCOGS,            "####
    p~ZCOGS_CUKY            "####
    FROM ZTOI_ORDER AS O INNER JOIN SCAL_TT_DATE AS T
    ON O~ZORDERDT EQ T~CALENDARDATE
    LEFT OUTER JOIN ZTOI_PRODUCT AS P
    ON O~ZPRODCO EQ P~ZPRODCO
        LEFT OUTER JOIN ZTOI_STACHK AS S
    ON O~ZORDERNUM EQ S~ZORDERNUM
        LEFT OUTER JOIN ZTOI_DELIVERY AS D
    ON O~ZORDERNUM EQ D~ZORDERNUM
        LEFT OUTER JOIN ZTOI_CONSUMER AS C
    ON O~ZCUSTOMID EQ C~ZCUSTOMID
    LEFT OUTER JOIN DD07V AS TX
    ON P~ZCATEGORY EQ TX~DOMVALUE_L
    AND TX~DOMNAME EQ 'ZDO15_CATE'
    AND TX~DDLANGUAGE EQ @SY-LANGU
    WHERE T~YEARMONTH IN @SO_SPMON
    INTO CORRESPONDING FIELDS OF TABLE @GT_DATA.

  SELECT
  ROW~YEARMONTH,
  ROW~ZCATEGORY,
  ROW~ZCATEGORY_TX,
  ROW~ZUNIT,
  ROW~ZAMOUNT_CUKY,
    row~ZCOGS_CUKY,   "####
  SUM( ROW~ZQUANTITY ) AS ZQUANTITY,  "##
  SUM( ROW~ZAMOUNT ) AS ZAMOUNT ,      "####
  SUM( ROW~ZCOGS ) AS ZCOGS       "####

  FROM @GT_DATA AS ROW
    GROUP BY ROW~YEARMONTH, ROW~ZCATEGORY, ROW~ZCATEGORY_TX, ROW~ZUNIT, ROW~ZAMOUNT_CUKY, row~ZCOGS_CUKY
     INTO CORRESPONDING FIELDS OF TABLE @GT_DATA_SUMMARY.

ENDFORM.
