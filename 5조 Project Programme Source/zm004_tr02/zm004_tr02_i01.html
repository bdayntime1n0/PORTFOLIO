MODULE USER_COMMAND_0100 INPUT.

 SAVEOK = OKCODE. "####

 CLEAR OKCODE.


CASE 'X'.
  WHEN P_EXTER.
  CASE SAVEOK.
    WHEN 'SAVE'.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.
    WHEN 'REFRESH'.
      PERFORM SELECT_EXTER.
      PERFORM REFRESH_TABLE_DISPLAY USING GO_GRID.
   ENDCASE.

  WHEN P_INTER.
  CASE SAVEOK.
    WHEN 'SAVE'.
    WHEN 'BACK'.
      LEAVE TO SCREEN 0.     "BACK# #, ## #### ####
    WHEN 'REFRESH'.
      PERFORM SELECT_INTER.
      PERFORM REFRESH_TABLE_DISPLAY USING GO_GRID.
   ENDCASE.

  ENDCASE.

ENDMODULE.
