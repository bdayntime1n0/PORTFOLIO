FORM SELECT_DISPLAY .

  REFRESH : GT_DISPLAY.

  SELECT  B~ZPLANT
         ,A~ZORDERNUM
         ,B~ZSORG
         ,B~ZSGRU
         ,A~ZCUSTOMID
         ,C~ZCUSTOMNM
         ,A~ZPRODCO
         ,B~ZPRODNM
         ,A~ZQUANTITY
         ,A~ZUNIT
         ,A~ZORDERDT
         ,A~ZAMOUNT
         ,A~ZAMOUNT_CUKY
         ,B~ZCOGS
         ,B~ZCOGS_CUKY
         ,D~ZPRFLAG
   FROM ZTOI_ORDER AS A
   LEFT OUTER JOIN ZTOI_PRODUCT  AS B ON A~ZPRODCO = B~ZPRODCO
   LEFT OUTER JOIN ZTOI_CONSUMER AS C ON A~ZCUSTOMID = C~ZCUSTOMID
   INNER JOIN ZTOI_STACHK   AS D ON A~ZORDERNUM = D~ZORDERNUM
*       INNER JOIN ZTOI_VENDOR   AS E ON A~ZVENDORCO = E~ZVENDORCO
   WHERE D~ZCANCHK NE 'X'
     AND D~ZSOFLAG IS INITIAL
   ORDER BY A~ZORDERNUM
   INTO CORRESPONDING FIELDS OF TABLE @GT_DISPLAY.

  IF GT_DISPLAY[] IS INITIAL.
    MESSAGE S000(OO) DISPLAY LIKE 'E' WITH '### #### #### ####.'.
  ELSE.
    MESSAGE S000(OO) WITH SY-DBCNT '## ## #####.'.
  ENDIF.

  IF SY-SUBRC EQ 0.

    LOOP AT GT_DISPLAY.

      GT_DISPLAY-STATUS = ICON_YELLOW_LIGHT.

      MODIFY GT_DISPLAY TRANSPORTING STATUS.

    ENDLOOP.

  ENDIF.

ENDFORM.

FORM SELECT_CUSTOM .

  SELECT C~ZPLANT,
         C~ZSORG,
         C~ZSGRU,
         A~ZORDERNUM,
         A~ZORDERDT,
         B~ZSOFLAG,
         A~ZCUSTOMID,
         C~ZPRODCO,
         C~ZPRODNM,
         A~ZQUANTITY,
         A~ZUNIT,
         A~ZAMOUNT,
         A~ZAMOUNT_CUKY,
         C~ZCOGS,
         C~ZCOGS_CUKY,
         B~ZPRFLAG,
         A~CRDAT,
         A~CRNAM
     FROM ZTOI_ORDER AS A
     LEFT OUTER JOIN ZTOI_STACHK AS B ON A~ZORDERNUM = B~ZORDERNUM
     INNER JOIN ZTOI_PRODUCT  AS C ON A~ZPRODCO = C~ZPRODCO
     LEFT OUTER JOIN VBEP AS D ON B~ZPRFLAG = D~BANFN
     WHERE B~ZSOFLAG   IN @S_SORDER
     AND   A~ZCUSTOMID IN @S_CUSTID
     AND   A~ZORDERDT  IN @S_SODATE
     AND   B~ZSOFLAG   IS NOT INITIAL
     ORDER BY B~ZSOFLAG DESCENDING
     INTO CORRESPONDING FIELDS OF TABLE @GT_CUSTOMER.

  IF GT_CUSTOMER[] IS INITIAL.
    MESSAGE S000(OO) DISPLAY LIKE 'E'
    WITH '#### ### Sales Order# #### ####.' .
  ELSE.
    MESSAGE S000(OO) WITH 'Sales Order' SY-DBCNT '## #######'.
  ENDIF.

ENDFORM.
