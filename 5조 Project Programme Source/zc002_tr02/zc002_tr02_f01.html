FORM GET_DATA .

  SELECT
    CASE
      WHEN B~ZPRODFLAG EQ 'X' THEN @ICON_LED_GREEN
      WHEN B~ZPRODFLAG EQ '01' THEN @ICON_LED_YELLOW
      ELSE @ICON_LED_RED END AS LIGHT,
    B~ZPRODFLAG,
    A~ZORDERNUM,
    A~ZCUSTOMID,
    A~ZPRODCO,
    C~ZPRODNM,
    A~ZVENDORCO,
    C~ZCATEGORY,
    C~ZPURCH_GRU,
    C~ZSGRU,
    A~ZQUANTITY,
    A~ZUNIT,
    A~ZAMOUNT,
    A~ZAMOUNT_CUKY,
    C~ZCOGS,
    C~ZCOGS_CUKY,
    A~ZORDERDT,
    A~ZDELVDT,
    A~ZTRANSNM,
    A~ZDELVCOMP
   FROM ZTOI_ORDER AS A LEFT OUTER JOIN ZTOI_STACHK AS B
    ON A~ZORDERNUM = B~ZORDERNUM
    LEFT OUTER JOIN ZTOI_PRODUCT AS C ON A~ZPRODCO = C~ZPRODCO
   INTO CORRESPONDING FIELDS OF TABLE @GT_DISP
   WHERE A~ZORDERNUM  IN @SO_ORDER
*    AND  A~ZORDERDT   eq @PO_DATE
*    AND  A~ZORDERDT   IN @SO_DATE
    AND  A~ZPRODCO   IN @SO_ZPROD
    ORDER BY light DESCENDING, A~ZORDERNUM DESCENDING.
  .
ENDFORM.

*&---------------------------------------------------------------------*
FORM USER_COMMAND_0100 .

  DATA(LV_UCOMM) = GV_UCOMM.
  CLEAR GV_UCOMM.

  CASE LV_UCOMM.
    WHEN 'CRMM'.
      PERFORM UCOMM_CRMM.
    WHEN 'CRCO'.
      PERFORM UCOMM_CRCO.
    WHEN 'REFRESH'.
      REFRESH GT_DISP.
      PERFORM GET_DATA.
    WHEN 'GOMM'.
      PERFORM UCOMM_GOMM.
    WHEN 'MULTIMM'.
      PERFORM UCOMM_MULTIMM.
    WHEN 'MULTICO'.
      PERFORM UCOMM_MULTICO.
    WHEN OTHERS.
  ENDCASE.

ENDFORM.

FORM EXIT.

  DATA(LV_UCOMM) = GV_UCOMM.
  CLEAR GV_UCOMM.

  CASE LV_UCOMM.
    WHEN 'EXIT'.
      LEAVE PROGRAM.
    WHEN 'BACK' OR 'CANC'.
      LEAVE TO SCREEN 0.
    WHEN OTHERS.
  ENDCASE.

ENDFORM.
