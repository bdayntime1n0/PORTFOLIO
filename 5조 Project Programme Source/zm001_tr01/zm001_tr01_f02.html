FORM DISPLAY_ALV_DATA USING PV_TNAME TYPE ANY
                            PV_LAYO       TYPE ANY
                      CHANGING PO_GRID    TYPE REF TO CL_GUI_ALV_GRID.

  CONSTANTS : LC_TB(2) VALUE '[]',
              LC_SP(1) VALUE SPACE.

  DATA : LV_TABLE LIKE FELD-NAME,
         LV_LAYOT LIKE FELD-NAME.

  FIELD-SYMBOLS <LT_TABLE>  TYPE STANDARD TABLE.
  FIELD-SYMBOLS <LT_LAYOT> TYPE ANY.

  CONCATENATE PV_TNAME LC_TB INTO LV_TABLE.
  ASSIGN (LV_TABLE) TO <LT_TABLE>.

  CONCATENATE PV_LAYO LC_SP INTO LV_LAYOT.
  ASSIGN (LV_LAYOT) TO <LT_LAYOT>.

  CALL METHOD PO_GRID->SET_TABLE_FOR_FIRST_DISPLAY
    EXPORTING
      IS_LAYOUT                     = <LT_LAYOT>
    CHANGING
      IT_FIELDCATALOG               = GT_FIELDCAT
      IT_OUTTAB                     = <LT_TABLE>
    EXCEPTIONS
      INVALID_PARAMETER_COMBINATION = 1
      PROGRAM_ERROR                 = 2
      TOO_MANY_LINES                = 3
      OTHERS                        = 4.

FORM GET_FIELDCATALOG_BY_FUNC TABLES PT_DATA PT_FIELDCAT USING PV_TABNAME.

  DATA : LT_FIELDCAT  TYPE  SLIS_T_FIELDCAT_ALV.
  DATA : LV_STR       TYPE  DD02L-TABNAME.

  CLEAR LV_STR.

  LV_STR = PV_TABNAME.

  CALL FUNCTION 'REUSE_ALV_FIELDCATALOG_MERGE'
    EXPORTING
      I_PROGRAM_NAME         = SY-REPID
      I_STRUCTURE_NAME       = LV_STR
      I_CLIENT_NEVER_DISPLAY = 'X'
      I_INCLNAME             = SY-REPID
    CHANGING
      CT_FIELDCAT            = LT_FIELDCAT[].

  IF SY-SUBRC EQ 0.

    CALL FUNCTION 'LVC_TRANSFER_FROM_SLIS'
      EXPORTING
        IT_FIELDCAT_ALV = LT_FIELDCAT
      IMPORTING
        ET_FIELDCAT_LVC = PT_FIELDCAT[]
      TABLES
        IT_DATA         = PT_DATA.

  ENDIF.

ENDFORM.
