FORM SELECTION_SCREEN_OUTPUT .

  LOOP AT SCREEN.

    CASE 'X'.
      WHEN P_CREATE.
        IF SCREEN-GROUP1 EQ 'B2'.
          SCREEN-ACTIVE = 0.
        ENDIF.
      WHEN P_EXCUTE.
        IF SCREEN-GROUP1 EQ 'B2'.
          SCREEN-ACTIVE = 1.
        ENDIF.
    ENDCASE.

    MODIFY SCREEN.

  ENDLOOP.

ENDFORM.

FORM SELECT_DATA .

  DATA : LR_CHK TYPE RANGE OF ZTOI_STACHK-ZSOFLAG WITH HEADER LINE.

  REFRESH GT_DATA.

  CASE 'X'.
    WHEN P_CREATE.

      LR_CHK-OPTION = 'EQ'.
      LR_CHK-SIGN   = 'I'.
      LR_CHK-LOW    = ''.

      APPEND LR_CHK.

  ENDCASE.

  SELECT
        A~ZORDERNUM
      , B~ZSOFLAG
      , B~ZPRFLAG
      , B~ZPOFLAG
      , C~ZCATEGORY
      , C~ZPRODCO
      , C~ZPRODNM
      , A~ZQUANTITY
      , C~ZCOGS
      , A~ZAMOUNT
      , D~ZVENDORCO
      , D~ZVENDORNM
      , C~ZPLANT
      , C~ZSORG
      , C~ZSGRU
      , C~ZPURCH_GRU
      , B~ZETC
    FROM ZTOI_ORDER AS A
    INNER JOIN ZTOI_STACHK AS B
            ON A~ZORDERNUM EQ B~ZORDERNUM
    LEFT OUTER JOIN ZTOI_PRODUCT AS C
                 ON A~ZPRODCO EQ C~ZPRODCO
    LEFT OUTER JOIN ZTOI_VENDOR AS D
                 ON A~ZVENDORCO EQ D~ZVENDORCO
              WHERE B~ZPRFLAG IS NOT INITIAL
                AND B~ZPOFLAG IN @LR_CHK
    ORDER BY A~ZORDERNUM DESCENDING
    INTO CORRESPONDING FIELDS OF TABLE @GT_DATA.

  IF SY-SUBRC EQ 0.

    LOOP AT GT_DATA.

      IF GT_DATA-ZPOFLAG IS NOT INITIAL.

        GT_DATA-STATUS = ICON_GREEN_LIGHT.

        MESSAGE S000(OO) WITH SY-DBCNT '# #######.'.

      ELSE.

        GT_DATA-STATUS = ICON_YELLOW_LIGHT.

        MESSAGE S000(OO) WITH SY-DBCNT '# #######.'.

      ENDIF.

      MODIFY GT_DATA TRANSPORTING STATUS.

    ENDLOOP.

  ELSE.

    GV_CHK = 'N'.

    IF P_CREATE EQ 'X'.

      MESSAGE I000(OO) WITH '### #### #### ####.'.

    ELSE.

      MESSAGE I000(OO) WITH '#### #### ####.'.

    ENDIF.

  ENDIF.

ENDFORM.

    MODIFY GT_DATA TRANSPORTING STATUS ZPOFLAG ZETC WHERE ZORDERNUM = GT_DATA-ZORDERNUM.

  ENDLOOP.

  PERFORM REFRESH_GRID USING GO_GRID1.
