FORM MODIFY_SCREEN .

  LOOP AT SCREEN.
    CASE 'X'.
      WHEN P_CC.
        IF SCREEN-GROUP1 = 'INQ'.
          SCREEN-INPUT  = '0'.
          SCREEN-ACTIVE = '0'.
        ENDIF.

      WHEN P_CINQ.
        IF SCREEN-GROUP1 = 'INQ'.
          SCREEN-INPUT  = '1'.
          SCREEN-ACTIVE = '1'.
        ENDIF.

    ENDCASE.

    MODIFY SCREEN.

  ENDLOOP.


ENDFORM.
