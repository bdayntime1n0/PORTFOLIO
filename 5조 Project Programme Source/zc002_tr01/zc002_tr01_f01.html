FORM CHK USING PV_CHK.

  SELECT A~ZORDERNUM, B~ZCUSTOMID, C~ZCUSTOMFLAG
    FROM ZTOI_ORDER AS A
    LEFT OUTER JOIN ZTOI_CONSUMER AS B
                 ON A~ZCUSTOMID EQ B~ZCUSTOMID
         INNER JOIN ZTOI_STACHK AS C
                 ON A~ZORDERNUM EQ C~ZORDERNUM
         INTO TABLE @DATA(LT_DATA)
              WHERE B~ZCUSTOMID EQ @PV_CHK.

  CHECK SY-SUBRC EQ 0.

  LOOP AT LT_DATA INTO DATA(LS_DATA).

    READ TABLE GT_DATA INTO GS_DATA WITH KEY ZCUSTOMID = LS_DATA-ZCUSTOMID.

    CHECK SY-SUBRC EQ 0.

    IF LS_DATA-ZCUSTOMFLAG IS INITIAL.
      GS_DATA-STATUS = ICON_YELLOW_LIGHT.
      GS_DATA-ZETC   = '#### ## ########!'.
      GS_DATA-LCOLOR = 'C300'.

      MODIFY GT_DATA FROM GS_DATA TRANSPORTING STATUS ZETC LCOLOR WHERE ZCUSTOMID = PV_CHK.

    ELSE.
      GS_DATA-STATUS = ICON_GREEN_LIGHT.

      MODIFY GT_DATA FROM GS_DATA TRANSPORTING STATUS WHERE ZCUSTOMID = PV_CHK.

    ENDIF.

  ENDLOOP.

ENDFORM.
