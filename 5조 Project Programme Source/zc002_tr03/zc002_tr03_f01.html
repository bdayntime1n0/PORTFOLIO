FORM MODIFY_SCREEN .

  CASE 'X'.
    WHEN P_V1.
      LOOP AT SCREEN.
        IF SCREEN-GROUP1 = 'DSP'.
           SCREEN-ACTIVE = '0'.
        ENDIF.

        MODIFY SCREEN.

      ENDLOOP.

  ENDCASE.

ENDFORM.

FORM SELECT_DATA .

  REFRESH GT_VMST.

  SELECT  A~ZORDERNUM,
          B~ZPURCH_GRU,
          B~ZPLANT,
          A~ZVENDORCO,
          C~ZVENDORNM,
          C~ZVPOSTNUM,
          C~ZVCITY,
          C~ZVADDRESS,
          B~ZCOGS_CUKY,
          A~CRDAT,
          A~CRNAM
    FROM ZTOI_ORDER AS A
    LEFT OUTER JOIN ZTOI_PRODUCT AS B
                 ON A~ZPRODCO = B~ZPRODCO
    LEFT OUTER JOIN ZTOI_VENDOR  AS C
                 ON A~ZVENDORCO = C~ZVENDORCO
    INNER JOIN ZTOI_STACHK  AS D
                 ON A~ZORDERNUM = D~ZORDERNUM
    WHERE D~ZVENDORFLAG IS INITIAL
    ORDER BY C~ZVENDORCO
    INTO CORRESPONDING FIELDS OF TABLE @GT_VMST[].

ENDFORM.

FORM SELECT_DISP .

  SELECT DISTINCT C~ZPLANT,
                  C~ZPURCH_GRU,
                  A~ZVENDORCO,
                  B~ZVENDORNM,
                  B~ZVADDRESS,
                  B~ZVCITY,
                  B~ZVPOSTNUM,
                  C~ZCOGS_CUKY,
                  D~LAND1,
                  D~ERDAT AS CRDAT ,
                  D~ERNAM AS CRNAM
        FROM  ZTOI_ORDER AS A
        LEFT OUTER JOIN ZTOI_VENDOR  AS B ON A~ZVENDORCO = B~ZVENDORCO
        LEFT OUTER JOIN ZTOI_PRODUCT AS C ON A~ZPRODCO = C~ZPRODCO
        INNER JOIN LFA1              AS D ON B~ZVENDORCO = D~LIFNR
        WHERE A~ZVENDORCO IN @S_VNCODE
        AND   B~ZVENDORNM IN @S_VNNMAE
        AND   D~ERDAT IN @S_CRDATE
        AND   D~ERNAM IN @S_UNAME
        ORDER BY A~ZVENDORCO DESCENDING
        INTO CORRESPONDING FIELDS OF TABLE @GT_SEARCH .

  IF SY-SUBRC EQ 0.
    MESSAGE S000(OO) WITH '###' SY-DBCNT '## #######'.
  ELSE.
    MESSAGE S000(OO) DISPLAY LIKE 'E' WITH '## ### ## #### ####'.
    EXIT.
  ENDIF.

ENDFORM.

    EXPORTING

FORM STATUS_CHK .

  SELECT SINGLE A~LIFNR, B~ZORDERNUM
    FROM LFA1 AS A
    INNER JOIN ZTOI_ORDER AS B
            ON A~LIFNR EQ B~ZVENDORCO
   WHERE A~LIFNR EQ @GT_S_DISPLAY-ZVENDORCO
    INTO @DATA(LS_DATA).

  IF SY-SUBRC EQ 0.

    UPDATE ZTOI_STACHK SET ZVENDORFLAG = 'X' WHERE ZORDERNUM EQ LS_DATA-ZORDERNUM.

  ENDIF.

ENDFORM.
